HMC5883L

Base class: I2CCompass

Manages a HMC5883L I2C Compass

Example Configuration

[HMC5883L= 8]
  name= "Compass"
  interval= 50
  bus= 0
  calibX= -6,0,6
  calibY= -6,0,6
  calibZ=-6,0,6
  trim= 180
  location= -1.8, 52, 100
  mode=1
  centre = -3.188, 4.167, -0.1043
  $location = @>GPS.location
  $roll = @>MPU6050.roll
  $pitch = @>MPU6050.pitch
  publish =heading, vector, calibX, calibY, calibZ
  publish = trim, mode, roll, pitch, raw
